# 4.7 Generating Lorem Text

## A. Concept Overview

### What & Why
Text content â€“ descriptions, comments, reviews, blog posts â€“ is essential for testing APIs that handle user-generated content. Faker generates **Lorem Ipsum** text and realistic sentences in various lengths, perfect for testing text fields, character limits, and content processing features.

### Analogy
Think of Faker's text methods like **a content generator with size options**.

Like ordering coffee:
- `word()` â†’ Espresso (single shot)
- `sentence()` â†’ Small (one sentence)
- `paragraph()` â†’ Medium (multiple sentences)
- `text()` â†’ Large (custom size)

Each size serves a different testing need!

"""Tests for text generation."""
import pytest
from faker import Faker


def test_word_generation():
    """Test single word generation."""
    fake = Faker()
    
    word = fake.word()
    
    # Should be a single word
    assert isinstance(word, str)
    assert " " not in word
    assert len(word) > 0
    
    print(f"âœ… Word: {word}")


def test_words_list():
    """Test multiple words generation."""
    fake = Faker()
    
    words = fake.words(5)
    
    # Should be list of 5 words
    assert isinstance(words, list)
    assert len(words) == 5
    assert all(isinstance(w, str) for w in words)
    
    print(f"âœ… Words list: {words}")


def test_sentence_generation():
    """Test sentence generation."""
    fake = Faker()
    
    sentence = fake.sentence()
    
    # Should be a string
    assert isinstance(sentence, str)
    
    # Should end with period
    assert sentence.endswith(".")
    
    # Should contain multiple words
    assert " " in sentence
    
    # Should have at least a few words
    word_count = len(sentence.split())
    assert word_count >= 3
    
    print(f"âœ… Sentence: {sentence}")


def test_sentence_with_word_count():
    """Test sentence with specific word count."""
    fake = Faker()
    
    sentence = fake.sentence(nb_words=10)
    
    # Should have approximately 10 words (give or take)
    word_count = len(sentence.replace(".", "").split())
    assert 8 <= word_count <= 12
    
    print(f"âœ… Sentence ({word_count} words): {sentence}")


def test_sentences_list():
    """Test multiple sentences generation."""
    fake = Faker()
    
    sentences = fake.sentences(3)
    
    # Should be list of 3 sentences
    assert isinstance(sentences, list)
    assert len(sentences) == 3
    assert all(isinstance(s, str) for s in sentences)
    assert all(s.endswith(".") for s in sentences)
    
    print(f"âœ… Sentences list:")
    for i, sent in enumerate(sentences, 1):
        print(f"   {i}. {sent}")


def test_paragraph_generation():
    """Test paragraph generation."""
    fake = Faker()
    
    paragraph = fake.paragraph()
    
    # Should be a string
    assert isinstance(paragraph, str)
    
    # Should contain multiple sentences
    assert paragraph.count(".") >= 2
    
    # Should be substantial
    assert len(paragraph) > 50
    
    print(f"âœ… Paragraph ({len(paragraph)} chars):")
    print(f"   {paragraph}")


def test_paragraph_with_sentence_count():
    """Test paragraph with specific sentence count."""
    fake = Faker()
    
    paragraph = fake.paragraph(nb_sentences=5)
    
    # Should have approximately 5 sentences
    sentence_count = paragraph.count(".")
    assert 4 <= sentence_count <= 6
    
    print(f"âœ… Paragraph ({sentence_count} sentences):")
    print(f"   {paragraph}")


def test_paragraphs_list():
    """Test multiple paragraphs generation."""
    fake = Faker()
    
    paragraphs = fake.paragraphs(3)
    
    # Should be list of 3 paragraphs
    assert isinstance(paragraphs, list)
    assert len(paragraphs) == 3
    assert all(isinstance(p, str) for p in paragraphs)
    
    print(f"âœ… Generated {len(paragraphs)} paragraphs!")


def test_text_with_max_chars():
    """Test text generation with character limit."""
    fake = Faker()
    
    text_50 = fake.text(max_nb_chars=50)
    text_200 = fake.text(max_nb_chars=200)
    
    # Should respect max length (approximately)
    assert len(text_50) <= 60  # Some margin
    assert len(text_200) <= 210  # Some margin
    
    print(f"âœ… Text with max chars!")
    print(f"   50 chars: {text_50}")
    print(f"   200 chars: {text_200[:50]}...")


def test_texts_list():
    """Test multiple texts generation."""
    fake = Faker()
    
    texts = fake.texts(nb_texts=3, max_nb_chars=50)
    
    # Should be list of 3 texts
    assert isinstance(texts, list)
    assert len(texts) == 3
    
    # Each should be under max length (approximately)
    assert all(len(t) <= 60 for t in texts)
    
    print(f"âœ… Generated {len(texts)} text blocks!")


# ==================== PRACTICAL USE CASES ====================

def test_blog_post_content():
    """Test generating blog post content."""
    fake = Faker()
    
    blog_post = {
        "title": fake.sentence(nb_words=6).replace(".", ""),  # Remove period
        "summary": fake.text(max_nb_chars=200),
        "content": "\n\n".join(fake.paragraphs(5)),
        "tags": fake.words(3)
    }
    
    # Validate structure
    assert len(blog_post["title"]) > 0
    assert len(blog_post["summary"]) <= 210
    assert blog_post["content"].count("\n\n") >= 3  # Multiple paragraphs
    assert len(blog_post["tags"]) == 3
    
    print(f"âœ… Blog post generated!")
    print(f"   Title: {blog_post['title']}")
    print(f"   Tags: {blog_post['tags']}")
    print(f"   Content length: {len(blog_post['content'])} chars")


def test_product_description():
    """Test generating product descriptions."""
    fake = Faker()
    
    product = {
        "name": " ".join(fake.words(2)).title(),
        "short_description": fake.sentence(),
        "long_description": fake.text(max_nb_chars=500),
        "features": fake.sentences(5)
    }
    
    # Validate
    assert len(product["name"]) > 0
    assert product["short_description"].endswith(".")
    assert len(product["long_description"]) <= 510
    assert len(product["features"]) == 5
    
    print(f"âœ… Product description generated!")
    print(f"   Name: {product['name']}")
    print(f"   Short: {product['short_description']}")


def test_user_review():
    """Test generating user review."""
    fake = Faker()
    
    review = {
        "title": fake.sentence(nb_words=5).replace(".", ""),
        "rating": fake.random_int(1, 5),
        "review_text": fake.paragraph(nb_sentences=3),
        "pros": fake.sentences(2),
        "cons": fake.sentences(2)
    }
    
    # Validate
    assert 1 <= review["rating"] <= 5
    assert len(review["review_text"]) > 50
    assert len(review["pros"]) == 2
    assert len(review["cons"]) == 2
    
    print(f"âœ… User review generated!")
    print(f"   Title: {review['title']}")
    print(f"   Rating: {review['rating']}/5")


def test_comment_thread():
    """Test generating comment thread."""
    fake = Faker()
    
    comments = []
    for i in range(5):
        comment = {
            "id": i + 1,
            "author": fake.user_name(),
            "text": fake.paragraph(nb_sentences=2),
            "created_at": fake.date_time_this_year()
        }
        comments.append(comment)
    
    assert len(comments) == 5
    
    print(f"âœ… Generated {len(comments)} comments!")
    for comment in comments[:2]:
        print(f"   {comment['author']}: {comment['text'][:40]}...")


def test_bio_generation():
    """Test generating user bio with length constraint."""
    fake = Faker()
    
    # Bio with max 150 characters
    bio = fake.text(max_nb_chars=150)
    
    assert len(bio) <= 160  # Some margin
    
    print(f"âœ… Bio ({len(bio)} chars): {bio}")


def test_search_keywords():

    assert len(keywords) == 5

## D. Text Methods Quick Reference

| Method | Returns | Use Case |
|--------|---------|----------|
| `word()` | String | Single word |
| `words(n)` | List | Multiple words |
| `sentence()` | String | One sentence |
| `sentences(n)` | List | Multiple sentences |
| `paragraph()` | String | One paragraph |
| `paragraphs(n)` | List | Multiple paragraphs |
| `text(max_nb_chars)` | String | Custom length text |
| `texts(nb_texts, max_nb_chars)` | List | Multiple texts |
